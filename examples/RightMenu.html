    var TST1 = [{"<img class='js_img' src='images/Transparent.gif' width='1' height='1'>javascript": function() { TstFunction('javascript'); }}, 
           {"<img class='goog_img' src='images/Transparent.gif' width='1' height='1'>Google": "http://www.google.com"},
