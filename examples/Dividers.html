Example of a menu using mouseover filters and dividers





Code

  • Javascript
  • HTML
  • includes 

            
    var Menu_w_Dividers = {  
                  'Search'  :'divider' , 
                  'Google'  :'http://www.google.com' , 
                  'Yahoo'  :'http://www.yahoo.com' , 
                  'Business'  :'divider' , 
                  'CNBC'  :'http://www.cnbc.com' , 
                  'MarketWatch' :'http://www.MarketWatch.com' 
    };

    $('button[id="menu_divs_filts"]').menuoptions({ 
        'MenuOptionsType': 'Navigate', 
        'Data': Menu_w_Dividers,
        'Filters': [{ 'Biz' : '^(CNBC|MarketWatch)'}, {'Search' :'^(Google|Yahoo)'} ], 
        'Sort': [], 
    }); 

    $("div#buttons").buttonset().css({"padding":"5px" });
   
        

            
                <div id=buttons>
                    <button type="button" id="menu_divs_filts" >Menu with dividers & filters</button>
